    fun consultarSaldo(senha: Int){

        if(senha == senhaCorreta){
            println("Saldo: R$$saldo")

            println("Senha Incorreta")
        }
    }
    
    
    fun deposito(senha: Int, valor: Double){
        
        if(senha == senhaCorreta){
            if(valor <= 0){
               println("deposito no R$$valor impossivel de ser realizado")

            saldo = valor
